Level Structure †

Ei = ∫IRi-ERi dt + [14]
    ERi = Ei/Pi
    IRi = gamma i*(Ii+k*Ti)*Si/Ni

FDM = ∫CFM dt + [DRR0]
DRR0 = Ii*n/UnitTime
    CFM = rhoM*(OAM-FDM)

FDR = ∫CFR dt + [DRR0]
    CFR = rhoR*(OAR-FDR)

FESi = ∫in dt + [0]
    in = RRMi+RRi

Ii = ∫ERi-RRi-ARi dt + [1]
    ARi = SRR/n
    RRi = Ii/TRi

MBM = ∫OAM-OFM dt + [DRR0*TOM]
TOM = 1
    OAM = ORR
    OFM = SRM

MBR = ∫OAR-OFR dt + [DRR0*TOR]
TOR = 1
    OAR = DRR
    OFR = (ARi+RRi)*n

MIM = ∫PRM-SRM dt + [DIM]
DIM = FDM*DCM
    PRM = DELAY3(ORM, TLM)
    SRM = MIN(DSM, MSM)

MIR = ∫PRR-SRR dt + [DIR]
DIR = FDR*DCR
    PRR = DELAY3(SRM, TLR)
    SRR = MIN(DSR, MSR)

MTM = ∫ORM-PRM dt + [DOM]
DOM = TLM*FDM
    ORM = MAX(0, FDM+AIM+AOM+ABM)

MTR = ∫SRM-PRR dt + [DRR0*TLR]
TLR = 3

Ri = ∫RRi+RRMi-WRi dt + [0]
    RRMi = Ti/TRMi
    WRi = Ri/TWi

Si = ∫WRi-IRi dt + [2688]
