Level Structure †

Ei = ∫IRi-ERi dt + [15]
    ERi = Ei/Pi
    IRi = gamma i*(Ii+k*Ti)*Si/Ni

FDR = ∫CFR dt + [DRR0]
DRR0 = Ii*n/UnitTime
    CFR = rhoR*(OAR-FDR)

FESi = ∫in dt + [0]
    in = RRMi+RRi

Ii = ∫ERi-RRi-ARi dt + [1]
    ARi = SRR/n
    RRi = Ii/TRi

MBR = ∫OAR-OFR dt + [DRR0*TOR]
TOR = 1
    OAR = DRR
    OFR = (ARi+RRi)*n

MIR = ∫PRR-SRR dt + [DIR]
DIR = FDR*DCR
    PRR = DELAY3(ORR, TLR)
